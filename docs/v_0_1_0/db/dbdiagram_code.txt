Table user {
  id uuid [primary key]
  
  login varchar
  password varchar
  
  mail varchar
  phone varchar
  
  first_name varchar
  second_name varchar
  date_of_birth date

  created_at timestamp
}

Table chat {
  id uuid [primary key]
}

Table chat_comment {
  id uuid [primary key]
  chat uuid [ref: > chat.id]
  user uuid [ref: > user.id]
}
